public class CameraActivity extends AppCompatActivity {

    private static final int REQUEST_CODE_PERMISSIONS = 10;
    private static final String[] REQUIRED_PERMISSIONS = new String[]{
            Manifest.permission.CAMERA,
            Manifest.permission.RECORD_AUDIO
    };

    private PreviewView viewFinder;
}

2Método onCreate()

@Override
protected void onCreate(Bundle savedInstanceState) {
    super.onCreate(savedInstanceState);
    setContentView(R.layout.activity_camera);

    viewFinder = findViewById(R.id.viewFinder);

    if (allPermissionsGranted()) {
        startCamera();
    } else {
        ActivityCompat.requestPermissions(
                this, REQUIRED_PERMISSIONS, REQUEST_CODE_PERMISSIONS);
    }
}

💻 Passo 2 - Código Java para Preview (2/2)

3Verificar Permissões

private boolean allPermissionsGranted() {
    for (String permission : REQUIRED_PERMISSIONS) {
        if (ContextCompat.checkSelfPermission(
                this, permission) != PackageManager.PERMISSION_GRANTED) {
            return false;
        }
    }
    return true;
}

@Override
public void onRequestPermissionsResult(int requestCode, 
        @NonNull String[] permissions, @NonNull int[] grantResults) {
    super.onRequestPermissionsResult(requestCode, permissions, grantResults);
    if (requestCode == REQUEST_CODE_PERMISSIONS) {
        if (allPermissionsGranted()) {
            startCamera();
        } else {
            Toast.makeText(this, "Permissões negadas", 
                    Toast.LENGTH_SHORT).show();
            finish();
        }
    }
}

🚀 Passo 3 - Implementar startCamera()

📹Inicializar Preview

private void startCamera() {
    ListenableFuture<ProcessCameraProvider> cameraProviderFuture =
            ProcessCameraProvider.getInstance(this);

    cameraProviderFuture.addListener(() -> {
        try {
            ProcessCameraProvider cameraProvider = 
                    cameraProviderFuture.get();

            // Selecionar câmera traseira
            CameraSelector cameraSelector = 
                    CameraSelector.DEFAULT_BACK_CAMERA;

            // Criar Preview
            Preview preview = new Preview.Builder().build();
            preview.setSurfaceProvider(
                    viewFinder.getSurfaceProvider());

            // Desvincular tudo antes de vincular novos use cases
            cameraProvider.unbindAll();
            
            // Vincular Preview ao lifecycle
            cameraProvider.bindToLifecycle(
                    this, cameraSelector, preview);

        } catch (ExecutionException | InterruptedException e) {
            e.printStackTrace();
        }
    }, ContextCompat.getMainExecutor(this));
}

📷 Passo 6 - Implementar takePhoto()

private void takePhoto() {
    if (imageCapture == null) return;

    // Definir diretório de saída
    File outputDir = getExternalFilesDir(Environment.DIRECTORY_PICTURES);
    if (outputDir == null) {
        outputDir = getFilesDir();
    }

    // Criar nome único para o arquivo
    String fileName = "IMG_" + System.currentTimeMillis() + ".jpg";
    File photoFile = new File(outputDir, fileName);

    // Configurar opções de saída
    ImageCapture.OutputFileOptions outputOptions =
            new ImageCapture.OutputFileOptions.Builder(photoFile).build();

    // Capturar foto
    imageCapture.takePicture(
            outputOptions,
            ContextCompat.getMainExecutor(this),
            new ImageCapture.OnImageSavedCallback() {
                @Override
                public void onImageSaved(
                        @NonNull ImageCapture.OutputFileResults results) {
                    Toast.makeText(CameraActivity.this,
                            "✅ Foto salva: " + photoFile.getAbsolutePath(),
                            Toast.LENGTH_LONG).show();
                }

                @Override
                public void onError(
                        @NonNull ImageCaptureException exception) {
                    Toast.makeText(CameraActivity.this,
                            "❌ Erro ao tirar foto",
                            Toast.LENGTH_SHORT).show();
                }
            }
    );
}

🎬 Passo 9 - Implementar startVideoRecording()

private void startVideoRecording() {
    if (videoCapture == null) return;

    // Definir diretório de saída
    File outputDir = getExternalFilesDir(Environment.DIRECTORY_MOVIES);
    if (outputDir == null) {
        outputDir = getFilesDir();
    }

    // Criar nome único para o arquivo
    String fileName = "VID_" + System.currentTimeMillis() + ".mp4";
    File videoFile = new File(outputDir, fileName);

    // Configurar opções de saída
    FileOutputOptions fileOutputOptions =
            new FileOutputOptions.Builder(videoFile).build();

    // Iniciar gravação
    currentRecording = videoCapture.getOutput()
            .prepareRecording(this, fileOutputOptions)
            .withAudioEnabled()  // Habilitar áudio
            .start(ContextCompat.getMainExecutor(this),
                    videoRecordEvent -> {
                        if (videoRecordEvent instanceof VideoRecordEvent.Start) {
                            Toast.makeText(this,
                                    "🔴 Gravação iniciada",
                                    Toast.LENGTH_SHORT).show();
                        } else if (videoRecordEvent instanceof 
                                VideoRecordEvent.Finalize) {
                            VideoRecordEvent.Finalize finalizeEvent = 
                                    (VideoRecordEvent.Finalize) videoRecordEvent;
                            if (!finalizeEvent.hasError()) {
                                Toast.makeText(this,
                                        "✅ Vídeo salvo: " + 
                                        videoFile.getAbsolutePath(),
                                        Toast.LENGTH_LONG).show();
                            } else {
                                Toast.makeText(this,
                                        "❌ Erro ao gravar vídeo",
                                        Toast.LENGTH_SHORT).show();
                            }
                            currentRecording = null;
                        }
                    });
}

⏹️ Passo 10 - Implementar stopVideoRecording()

private void stopVideoRecording() {
    if (currentRecording != null) {
        currentRecording.stop();
        currentRecording = null;
    }
}
💡 Simples assim! O método stop() finaliza a gravação e dispara o evento VideoRecordEvent.Finalize que definimos anteriormente.
